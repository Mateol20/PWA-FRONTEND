¿Cómo se puede combinar .map() con .filter() para transformar y filtrar datos?
const productos = [
    { nombre: 'Laptop', precio: 1000, oferta: false },
    { nombre: 'Teclado', precio: 50, oferta: true },
    { nombre: 'Mouse', precio: 25, oferta: true }
];

const nombresEnOferta = productos
    .filter(p => p.oferta)         // Paso 1: Filtra solo los que están en oferta
    .map(p => p.nombre);           // Paso 2: Extrae solo el nombre

// Resultado: ["Teclado", "Mouse"]
